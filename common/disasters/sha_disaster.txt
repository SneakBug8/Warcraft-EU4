sha_of_doubt_disaster = {
	potential = {
		NOT = { has_country_flag = complete_stasis }
		capital_scope = {
			continent = pandaria
		}
	}


	can_start = {
		is_at_war = yes
		NOT = { war_score = -25 }
	}
	
	progress = {
		modifier = {
				factor = 1
				NOT = { war_score = -25 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -45 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -65 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -85 }
		}
	}
	
	can_stop = {
		NOT = { war_score = 1 }
	}
	
	can_end = {
		is_at_war = no
	}
	
	modifier = {
	
	}

	#on_start = 0
	#on_end = 0
	
	on_monthly = {
        random_events = { 
                35 = 0
                1 = sha.1	
        }
	}
}

sha_of_fear_disaster = {
	potential = {
		NOT = { has_country_flag = complete_stasis }
		capital_scope = {
			continent = pandaria
		}
	}


	can_start = {
		any_neighbor_country = {
			is_rival = ROOT
			num_of_cities = ROOT
			army_size = ROOT
		}
	}
	
	progress = {
		modifier = {
				factor = 1
			any_neighbor_country = {
				is_rival = ROOT
				num_of_cities = ROOT
				army_size = ROOT
			}
		}
		modifier = {
				factor = 1
				any_neighbor_country = {
					is_rival = ROOT
					num_of_cities = ROOT
					war_with = ROOT
				}
		}
		modifier = {
				factor = 1
				NOT = { has_opinion = {
					who = ROOT
					value = -150
				} }
		}
	}
	
	can_stop = {
		any_neighbor_country = {
			is_rival = ROOT
			OR = {
				NOT = { num_of_cities = ROOT }
				NOT = { army_size = ROOT }
			}
		}
	}
	
	can_end = {
		any_neighbor_country = {
			is_rival = ROOT
			OR = {
				NOT = { num_of_cities = ROOT }
				NOT = { army_size = ROOT }
			}
		}
	}
	
	modifier = {
	
	}

	#on_start = 0
	#on_end = 0
	
	on_monthly = {
        random_events = { 
                35 = 0
                1 = sha.4	
        }
	}
}
