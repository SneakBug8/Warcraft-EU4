# Dreadmaul
country_event = {
	id = wwu_setup_ogre.1
	title = wwu_setup_ogre.1.title
	desc = wwu_setup_ogre.1.desc
	picture = OGRE_MAGE_eventPicture

    major = yes
    
	trigger = {
        ai = no
        NOT = { exists = DRD }
        NOT = { has_global_flag = setup_dreadmaul }
        is_year = 620 # 28 years after game start
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_dreadmaul
        }
    }
    
    mean_time_to_happen = {
		months = 240
	}

	option = {
		name = wwu_setup_ogre.1.option.a
        
        area_blasted_lands_3 = {
            cede_province = DRD
            add_core = DRD
            change_religion = dreadmaul
            change_culture = geomancy
        }
        
        ai_chance = {
            factor = 100
        }
	}
	option = {
		name = wwu_setup_ogre.1.option.b
        
        trigger = {
            area_blasted_lands_3 = {
                type = any
                owned_by = ROOT
            }
        }
        
        area_blasted_lands_3 = {
            change_religion = dreadmaul
            change_culture = geomancy
        }
        
        ai_chance = {
