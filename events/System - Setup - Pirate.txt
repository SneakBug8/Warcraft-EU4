namespace = wwu_setup_pirate

# Southsea Freebooters
country_event = {
	id = wwu_setup_pirate.1
	title = wwu_setup_pirate.1.title
	desc = wwu_setup_pirate.1.desc
	picture = BRAWL_eventPicture

    major = yes
    
	trigger = {
        ai = no
        NOT = { exists = SSP }
        NOT = { has_global_flag = setup_southsea_freebooters }
        has_global_flag = setup_scourge
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_southsea_freebooters
        }
    }
    
    mean_time_to_happen = {
		months = 240
	}

	option = {
		name = wwu_setup_pirate.1.option.a
        
        area_tanaris_5 = {
            cede_province = SSP
            add_core = SSP
            change_religion = southsea_freebooter
            change_culture = avarice
        }
        
        ai_chance = {
            factor = 100
        }
	}
	option = {
		name = wwu_setup_pirate.1.option.b
        
        trigger = {
            area_tanaris_5 = {
                type = any
                owned_by = ROOT
            }
        }
        
        area_tanaris_5 = {
            change_religion = southsea_freebooter
            change_culture = avarice
        }
        
        ai_chance = {
            factor = 0
        }
	}
}
