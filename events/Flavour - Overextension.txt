# Merchants Resented
country_event = {
	id = wwu_overextension.6
	title = wwu_overextension.6.title
	desc = wwu_overextension.6.desc
	picture = MERCHANTS_TALKING_eventPicture

	is_triggered_only = yes
	
	trigger = {
		num_of_merchants = 1
		any_active_trade_node = {
			has_merchant = PREV
			NOT = { 
				has_trade_modifier = {
					who = ROOT
					name = merchant_recalled
				}
			}	
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_overextension.6.option.a
		
		random_active_trade_node = {
			limit = {
				NOT = { 
					has_trade_modifier = {
						who = ROOT
						name = merchant_recalled
					}
				}				
				has_merchant = PREV
			}
			if = {
				limit = {
					trade_share = {
						country = ROOT
						share = 5
					}
				}
				add_trade_node_income = -10
			}
			add_trade_modifier = {
				who = root
				duration = 3650
				power = -10
				key = merchant_recalled
			}
		}
	}
}

	id = wwu_overextension.8
	title = wwu_overextension.8.title
	desc = wwu_overextension.8.desc
	picture = COURT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		legitimacy_or_horde_unity = 15
		government = monarchy
		is_lesser_in_union = no
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = wwu_overextension.8.option.a
		
		if = {
			limit = { NOT = { is_warband_government_type = yes } }
			add_legitimacy = -15
		}
		if = {
			limit = { is_warband_government_type = yes }
			add_horde_unity = -15
		}
	}
}

# Advisor Resigns
country_event = {
	id = wwu_overextension.10
	title = wwu_overextension.10.title
	desc = wwu_overextension.10.desc
	picture = OVEREXTENSION_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			advisor = philosopher
			advisor = natural_scientist
			advisor = artist
			advisor = statesman
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_overextension.10.option.a
		
		reduce_meritocracy_effect = yes
		if = {
			limit = {
				advisor = philosopher
			}
			remove_advisor = philosopher
		}
		if = {
			limit = {
				advisor = natural_scientist
			}
			remove_advisor = natural_scientist
		}
		if = {
			limit = {
				advisor = artist
			}
			remove_advisor = artist
		}
		if = {
			limit = {
				advisor = statesman
			}
			remove_advisor = statesman
		}
	}
}
